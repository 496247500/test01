练习使用git命令
git add ./-u/-all  <filename>
git commit -m '提交注释'
git push  <remote> <remote>:<deractory>


git fetch  对比远程分支与本地分支
git checkout 切换分支
git pull   拉远程分支并合并本地分支
git merge  合并分支

git clone  复制远程仓库
git status -s  查看暂存仓库状态
git log  查看提交日志
git branch  查看本地分支
git branch <分支名字> 创建新的分支
git branch -d 删除本地分支
git push origin --delete remoteBranchName  删除远程分支（不要带origin）
